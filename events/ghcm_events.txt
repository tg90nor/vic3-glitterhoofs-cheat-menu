ghcm.32 = { # add attack
	type = country_event
	hidden = yes
	immediate = {
		if = {
			limit = {
				NOT = {
					has_global_variable = ghcm_offense_mult
				}
			}
			set_global_variable = {
				name = ghcm_offense_mult
				value = @ghcm_offense_increment
			}
		}
		else = {
			change_global_variable = {
				name = ghcm_offense_mult
				add = @ghcm_offense_increment
			}
		}
		remove_modifier = ghcm_add_attack
		if = {
			limit = {
				global_var:ghcm_offense_mult > 0
			}
			add_modifier = {
				name = ghcm_add_attack
				multiplier = global_var:ghcm_offense_mult
			}
		}
		else_if = {
			limit = {
				global_var:ghcm_offense_mult < 0
			}
			add_modifier = {
				name = ghcm_add_attack
				multiplier = global_var:ghcm_offense_mult
			}
		}
	}
}
